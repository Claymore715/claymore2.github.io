一.无根图转化为有根图

#include
#include

using namespace std;

#define MAX 1000001

vector G[MAX];
int P[MAX];

void inputTree(int n)
{
int u, v; //存放两个节点编号

//输入边信息
for (int i = 0; i < n-1; ++i)
{
    cin >> u >> v;
    G[u].push_back(v);
    G[v].push_back(u);
}

}

//u为当前节点,fa为它的父节点
void dfs(int u, int fa)
{a
int d = G[u].size();

for (int i = 0; i < d; ++i)
{
    int v = G[u][i];

    if (v != fa)
        dfs(v, P[v] = u);
}

}

int main()
{
int n; //节点数
int root; //根节点编号
cin >> n;
inputTree(n);

cin >> root;
P[root] = -1;
dfs(root, P[root]);

//输出
for (int i = 0; i < n; ++i)
{
    cout << "R[" << i << "] : " << P[i] << endl;
}

return 0;

}
